/**
 * @param {ListNode} head
 * @return {ListNode}
 */
var middleNode = function(head) {
    const nodeArray = [];
    let currentNode = head;
    while(currentNode) {
        nodeArray.push(currentNode);
        currentNode = currentNode.next;
    }
    
    return nodeArray[Math.floor(nodeArray.length/2)];
};
